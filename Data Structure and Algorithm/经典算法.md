### 带哈希表的斐波那契函数

```python3
def fib(n):
    res_dict = dict()  # 每个不同的n只计算一次存入字典
    return fib_helper(n,res_dict)

def fib_helper(n,res_dict):
    if n <=2:
        # print(1)
        return 1
    else:
        if n-1 in res_dict:
            fib1= res_dict[n-1]
        else:
            fib1 = fib_helper(n-1,res_dict)
            res_dict[n-1]=fib1
        if n-2 in res_dict:
            fib2 = res_dict[n-2]
        else:
            fib2 = fib_helper(n-2,res_dict)
            res_dict[n - 2] = fib2
        # print(fib1+fib2)
        return fib1+fib2
```
